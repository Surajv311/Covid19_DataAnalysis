countries = list(corona_dataset_aggregated.index) # this will return the index of the list 
max_infection_rates=[]
for c in countries:
    max_infection_rates.append(corona_dataset_aggregated.loc[c].diff().max())
max_infection_rates #now we have calculated the max inf rates in each country using a for loop
# its better to put these values as a col in the dataframe 
corona_dataset_aggregated["max_infection_rate"] = max_infection_rates
